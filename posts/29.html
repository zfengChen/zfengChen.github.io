面试高频题系列 — 作用域

全局作用域

全局作用于指的是最外层的作用域，类似于文件里的根目录。在全局作用域内声明的变量称为全局变量

var a = 'aaa' // 全局变量
function show(){
	var a = 'bbb'
}
console.log(a) // 输出 aaa

局部作用域

局部作用域指的是在函数内部的作用域，类似于文件里的根目录下的子目录。除此之外，在全局作用域内声明的变量称为全局变量

var a = 'aaa'
function show(){
	var a = 'bbb' // 局部变量
	console.log(a) // 输出 bbb
}

块级作用域

一般指的是if、while、for循环里面的作用域

var a = 'aaa'

if(true){
	var a = 'bbb'
}

console.log(a) // 输出 bbb

作用域提升

var可以将变量提升

function show(){
	console.log(a) // 输出 undefind
	var a = 'aaa'
	console.log(a) // 输出 aaa
}
show()

以上代码可以理解为在show()函数内，定义了一个变量a，但是并给a没有赋值

function show(){
	var a;
	console.log(a) // 输出 undefind
	a = 'aaa'
	console.log(a) // 输出 aaa
}
show()

函数声明提升

var a = 'bbb'
function show(){
	console.log(a) // 输出 undefind
	var a = 'aaa'
	console.log(fn()) // 输出 aaa
	function fn(){
		return a
	}
}

以上代码可以理解为

var a = 'bbb'
function show(){
	function fn(){
		return a
	}
	var a
	console.log(a) // 输出 undefind
	a = 'aaa'
	console.log(fn()) // 输出 aaa

}

作用域链

作用域链指的是js规定好的变量的访问次序。就好像以下例子，在show()函数内打印的a，会先查看函数内部是否存在a，没有话就在全局寻找a

var a = 'aaa'
var b = 'bbb'

function show(){
	var a = 'aaa1'
	console.log(a) // aaa1
	console.log(b) // bbb
}

show()
console.log(a) // aaa
